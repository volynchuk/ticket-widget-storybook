const AutoLoop = () => {
    const [state, setState] = useState('loading')
    useEffect(() => {
      const id = setInterval(() => {
        setState(prev => (prev === 'loading' ? 'loaded' : 'loading'))
      }, 2500)
      return () => clearInterval(id)
    }, [])

    return (
        <TicketWidget
            value="10,09%"
            direction="down"
            pulse={false}
            widthDur={220}
            fadeDur={180}
            charStagger={20}
            blurAmt={4}
            digitInterval={800}
            digitJitter={300}
            state={state} />
    );
};
